    ValidateDatabaseSchema
        ValidateDatabaseHasTables
        ValidateDatabaseTables
            <foreach table>
                ValidateTableHasColumns
                ValidateTableColumns
                ValidateTableHasPrimaryKey
                ValidatePriamryKey
                <if table has foreign keys>
                    ValidateForeignKeys
                        <foreach foreign key>
                            ValidateForeignKeyHasReferencedTable
                            ValidateForeignKeyReferencedTable
                            ValidateForeignKeyHasColumns
                            ValidateForeignKeyColumns
    ValidateGraphQLTypes
        <foreach type>
            <if type is paginated>
                ValidatePaginationTypeSchema
            <else>
                *ValidateTypeTable
                    ValidateTypeTableMatchesSchemaType
                ValidateGraphQLTypeFields
                    <foreach field>
                        ValidateFieldSchema
                        <case OneToOne> *ValidateOneToOneField
                        <case OneToMany> *ValidateOneToManyField
                        <case ManyToOne> *ValidateManyToOneField
                        <case ManyToMany> *ValidateManyToManyField
                        <case None> *ValidateNoneField
                    *ValidateNoUnmatchedSchemaFields
        ValidateNoUnmatchedTypes
    ValidateMutationResolvers
        <foreach mutation resolver>
            *ValidateResolverId
            *ValidateMutationTable
            <case insert mutation> *ValidateInsertMutationSchema
            <case unpdate mutation> *ValidateUpdateMutationSchema
        ValidateNoUnMatchedMutations
    ValidateQuery
        <foreach query field>
            ValidateFieldSchema
